  Id INTEGER PRIMARY KEY,
  FirstName TEXT,
  LastName TEXT,
  Patronymic TEXT,
  Age INTEGER,
  Country TEXT,
  YearOfOlimpiad INTEGER,
  KindOfSport TEXT,
  Place INTEGER,
  Dope TEXT
  );
